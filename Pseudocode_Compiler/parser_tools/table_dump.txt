
1. program ::= Program identifier decl_section stmt_section End.

// ─── Declaration ────────────────────────────────────────────────────────────

2. decl_section  ::= Declaration_Section decl_list
3. decl_list     ::= declaration decl_list   (3a)
                  | ε                              (3b)
4. declaration   ::= type id_list (4a)
                  | constant type identifier is literal (4b)
                  | define identifier as type (4c)  // FIX: as quoted as keyword

// ─── Statements ─────────────────────────────────────────────────────────────

5. stmt_section  ::= Statement_Section stmt_list

6. stmt_list     ::= statement stmt_list (6a)
                  | statement (6b)

// FIX: Removed scope_stmt_list — unified under stmt_list.
// FIX: Removed separate loop_stmt_list — see loop_item below.

7. statement     ::= action_stmt

8. action_stmt   ::= assignment    (8a)
                  | io_stmt          (8b)
                  | conditional      (8c)
                  | loop             (8d)
                  | switch           (8e)
                  | scope_statement  (8f)
                  | loop_control     (8g)   // FIX: break/continue visible at all nesting levels

8.5 scope_statement ::= scope block stmt_list done

// ─── Assignment ─────────────────────────────────────────────────────────────

9. assignment    ::= set identifier to expression opt_as_type (9a)
                  | set item expression of identifier to expression (9b)

10. opt_as_type   ::= as type (10a)
                  | ε (10b)


// ─── I/O ────────────────────────────────────────────────────────────────────

11. io_stmt       ::= say expression_list   (11a)
                  | read id_list            (11b)

12. expression_list ::= expression                          (12a)
                    | expression , expression_list    (12b)


// ─── Conditional ────────────────────────────────────────────────────────────

13. conditional   ::= if expression then stmt_list
                    else_if_clause else_clause done

14. else_if_clause ::= else_if expression then stmt_list else_if_clause (14a)
                   | ε (14b)

15. else_clause   ::= else stmt_list (15a)
                  | ε (15b)


// ─── Loops ──────────────────────────────────────────────────────────────────

16. loop          ::= while_loop   (16a)
                  | for_loop     (16b)
                  | repeat_loop  (16c)

17. while_loop    ::= while expression do stmt_list done
18. for_loop      ::= for_every identifier in range_spec do stmt_list done
19. repeat_loop   ::= repeat stmt_list until expression

// FIX: loop_stmt_list removed; stmt_list is used directly above.
// loop_control is now a valid action_stmt everywhere, including inside
// scope_statement blocks nested within loops.

20. range_spec    ::= range expression to expression (20a)
                  | identifier (20b)

21. loop_control  ::= break (21a)
                  | continue (21b)


// ─── Switch ─────────────────────────────────────────────────────────────────

22. switch        ::= consider expression case_list opt_otherwise done
23. case_list     ::= case literal then stmt_list case_list (23a)
                  | ε (23b)
24. opt_otherwise ::= otherwise stmt_list (24a)
                  | ε (24b)


// ─── Expressions (LALR(1) precedence tower) ──────────────────────────────────
//
// FIX: The original grammar had all logical operators as flat alternatives
// under expression, causing reduce/reduce conflicts. They are now a linear
// chain of left-recursive rules (lowest → highest precedence).

25. expression      ::= logic_or_expr

26. logic_or_expr   ::= logic_or_expror logic_and_expr (26a) 
                    | logic_and_expr (26b)

27. logic_and_expr  ::= logic_and_exprand logic_xor_expr (27a) 
                    | logic_xor_expr (27b)

28. logic_xor_expr  ::= logic_xor_expr xor logic_nand_expr (28a) 
                    | logic_nand_expr (28b)

// nand / nor: binary only, left-recursive for parsing purposes.
// Note: enforce non-associativity semantically if desired.
29. logic_nand_expr ::= logic_nand_expr nand logic_nor_expr (29a)
                    | logic_nor_expr (29b)

30. logic_nor_expr  ::= logic_nor_expr nor logic_not_expr (30a)
                    | logic_not_expr (30b)

31. logic_not_expr  ::= not logic_not_expr (31a)
                    | relational (31b)


// ─── Relational ─────────────────────────────────────────────────────────────
//
// FIX: Removed the separate special_string alternative (38b).
// special_string derived identifier, which also derives from
// arithmetic_expr, making 38a vs 38b ambiguous on a single lookahead token.
// String vs numeric operand validation moves to semantic analysis.

32. relational    ::= arithmetic_expr rel_op arithmetic_expr   (32a)
                  | arithmetic_expr                               (32b)

33. rel_op        ::= is (33a) | is_not (33b)
                  | equal_to (33c) | not_equal_to (33d)
                  | greater_than (33e) | less_than (33f)
                  | greater_than_or_equal_to (33g) | less_than_or_equal_to (33h)


// ─── Arithmetic ─────────────────────────────────────────────────────────────

34. arithmetic_expr ::= arithmetic_expr add_op term   (34a)
                    | term                               (34b)

35. add_op  ::= plus (35a) | minus (35b)

36. term    ::= term mult_op power   (36a)
            | power                    (36b)

37. mult_op ::= times (37a) | divided_by (37b) | modulo (37c)

38. power   ::= primary raised_to power   (38a)   // right-associative
            | primary                        (38b)


// ─── Primary ─────────────────────────────────────────────────────────────────
//
// FIX: item access only here. Removed duplicate item_op from
// special_literal, which created two parse paths for the same syntax.

39. primary ::= ( expression )                     (39a)
            | identifier                                 (39b)
            | literal                                    (39c)
            | item arithmetic_expr of special_list (39d)


// ─── Literals ────────────────────────────────────────────────────────────────

40. list_literal  ::= [ list_elements ]
41. list_elements ::= expression list_tail (41a)
                  | ε (41b)
42. list_tail     ::= , expression list_tail (42a)
                  | ε (42b)

43. literal       ::= list_literal (43a)
                  | num_literal      (43b)
                  | string_literal   (43c)
                  | boolean_literal  (43d)
                  | special_literal  (43e)

// FIX: item_op removed. item access is handled solely by primary 48d.
44. special_literal ::= length_op   (44a)
                    | find_op     	(44b)
                    | string_op   	(44c)

45. length_op ::= length_of special_string
46. find_op   ::= find special_string in special_string
47. string_op ::= join special_string with special_string

48. special_string ::= string_literal   (48a)
                   | identifier           (48b)


// ─── Misc ────────────────────────────────────────────────────────────────────

49. special_list  ::= identifier (49a) | list_literal (49b)

50. id_list       ::= identifier              (50a)
                  | identifier , id_list    (50b)

51. identifier    ::= letter id_tail
52. id_tail       ::= letter id_tail
                  | digit id_tail
                  | _ id_tail
                  | ε


// ─── Number literals ─────────────────────────────────────────────────────────

53. num_literal     ::= integer_literal   (53a)
                    | double_literal        (53b)
54. double_literal  ::= integer_literal . digit_seq
55. integer_literal ::= digit_seq | - digit_seq
56. digit_seq       ::= digit digit_seq | digit


// ─── String literals ─────────────────────────────────────────────────────────

57. string_literal  ::= '' string_content ''
58. string_content  ::= string_element string_content | ε
59. string_element  ::= string_char | escape_seq
60. escape_seq      ::= \\n | \\t | \\\ | \\\\
61. string_char     ::= [any printable ASCII except  or \ or newline]


// ─── Terminals ───────────────────────────────────────────────────────────────

62. boolean_literal ::= true | false
63. type    ::= integer | double | string | boolean | list
64. letter  ::= a … z | A … Z
65. digit   ::= 0 … 9
